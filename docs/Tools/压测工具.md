## k6

```shell
k6 run script.js
# 模拟10个虚拟用户(VU)，连续压测30秒 --iterations=10000
k6 run --vus 10 --duration 30s script.js
```

```js
import http from 'k6/http';
import { sleep } from 'k6';
export const options = {
  vus: 10,
  duration: '30s',
};
export default function () {
  http.get('http://test.k6.io');
  sleep(1);
}

```
